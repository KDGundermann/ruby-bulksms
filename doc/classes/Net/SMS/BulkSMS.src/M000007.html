                        def host(country)
                                case country
                                        when 'uk'
                                                'www.bulksms.co.uk'
                                        when 'usa'
                                                'usa.bulksms.com'
                                        when 'international'
                                                'bulksms.vsms.net'
                                        when 'safrica'
                                                'bulksms.2way.co.za'
                                        when 'spain'
                                                'bulksms.com.es'
                                end
                        end
