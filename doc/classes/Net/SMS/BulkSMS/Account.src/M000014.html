                                def credits
                                        Net::HTTP.start(host(@country), SERVICE_PORT) do |http|
                                                response = http.post SERVICE_PATH, self.to_http_query
                                                if response.body.include?('|')
                                                        rsp = Response.parse(response.body)
                                                        raise AccountError, rsp.description, caller
                                                end
                                                response.body.to_f
                                        end
                                end
