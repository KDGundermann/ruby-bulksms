                                def send_message(msg)
                                        payload = [@account.to_http_query, msg.to_http_query].join('&')
                                        Net::HTTP.start(host(@country), MESSAGE_SERVICE_PORT) do |http|
                                                resp = http.post(MESSAGE_SERVICE_PATH, payload)
                                                Response.parse(resp.body)
                                        end
                                end
